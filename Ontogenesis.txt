/**

 * @description NextAuth.js configuration with credentials and OAuth providers
 * Implements JWT strategy with encrypted tokens

import { NextAuthOptions } from 'next-auth';
import { PrismaAdapter } from '@auth/prisma-adapter';
import CredentialsProvider from 'next-auth/providers/credentials';
import GoogleProvider from 'next-auth/providers/google';
import bcrypt from 'bcryptjs';
import { z } from 'zod';
import prisma from './db';

/**
 * Schema for credential validation
 */
const credentialsSchema = z.object({
  email: z.string().email(),
  password: z.string().min(8),
});

/**
 * NextAuth configuration object
 */
export const authOptions: NextAuthOptions = {
  adapter: PrismaAdapter(prisma),
  session: {
    strategy: 'jwt',
    maxAge: 30 * 24 * 60 * 60, // 30 days
  },
  pages: {
    signIn: '/login',
    error: '/login',
  },
  providers: [
    GoogleProvider({
      clientId: process.env.GOOGLE_CLIENT_ID ?? '',
      clientSecret: process.env.GOOGLE_CLIENT_SECRET ?? '',
    }),
    CredentialsProvider({
      name: 'credentials',
      credentials: {
        email: { label: 'Email', type: 'email' },
        password: { label: 'Password', type: 'password' },
      },
      async authorize(credentials) {
        try {
          const parsed = credentialsSchema.safeParse(credentials);
          
          if (!parsed.success) {
            throw new Error('Invalid credentials format');
          }

          const { email, password } = parsed.data;

          const user = await prisma.user.findUnique({
            where: { email },
          });

          if (!user || !user.password) {
            throw new Error('User not found');
          }

          const isValid = await bcrypt.compare(password, user.password);

          if (!isValid) {
            throw new Error('Invalid password');
          }

          return {
            id: user.id,
            email: user.email,
            name: user.name,
            image: user.image,
          };
        } catch (error) {
          console.error('Auth error:', error);
          return null;
        }
      },
    }),
  ],
  callbacks: {
    async jwt({ token, user }) {
      if (user) {
        token.id = user.id;
      }
      return token;
    },
    async session({ session, token }) {
      if (token && session.user) {
        session.user.id = token.id as string;
      }
      return session;
    },
  },
  events: {
    async signIn({ user, isNewUser }) {
      console.log(`User ${user.email} signed in. New user: ${isNewUser}`);
    },
    async createUser({ user }) {
      console.log(`New user created: ${user.email}`);
    },
  },
};

import { type ClassValue, clsx } from 'clsx';

/**
 * Merge Tailwind CSS classes with proper precedence

  return twMerge(clsx(inputs));

/**
 * Format frequency to human-readable string
 */
export function formatFrequency(hz: number): string {
  if (hz >= 1000) {
    return `${(hz / 1000).toFixed(2)} kHz`;
  }
  return `${hz.toFixed(2)} Hz`;
}

/**
 * Format duration in seconds to MM:SS
 */
export function formatDuration(seconds: number): string {
  const mins = Math.floor(seconds / 60);
  const secs = Math.floor(seconds % 60);
  return `${mins.toString().padStart(2, '0')}:${secs.toString().padStart(2, '0')}`;
}

/**
 * Generate a cryptographically secure random ID
 */
export function generateId(): string {
  return `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
}

/**
 * Deep clone an object
 */
export function deepClone<T>(obj: T): T {
  return JSON.parse(JSON.stringify(obj));
}

/**
 * Debounce function execution
 */
export function debounce<T extends (...args: unknown[]) => unknown>(
  fn: T,
  delay: number
): (...args: Parameters<T>) => void {
  let timeoutId: ReturnType<typeof setTimeout>;
  
  return (...args: Parameters<T>) => {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(() => fn(...args), delay);
  };
}

/**
 * Calculate golden ratio proportions
 */
export function goldenRatio(base: number, power: number = 1): number {
  const PHI = 1.618033988749895;
  return base * Math.pow(PHI, power);
}

/**
 * Clamp value between min and max
 */
export function clamp(value: number, min: number, max: number): number {
  return Math.min(Math.max(value, min), max);
}

/**
 * Linear interpolation
 */
export function lerp(start: number, end: number, t: number): number {
  return start * (1 - t) + end * t;
}

/**
 * Exponential easing
 */
export function easeInOutExpo(t: number): number {
  return t === 0 ? 0 : t === 1 ? 1 : t < 0.5 
    ? Math.pow(2, 20 * t - 10) / 2 
    : (2 - Math.pow(2, -20 * t + 10)) / 2;
}

  leftChannel: Float32Array;
  rightChannel: Float32Array;
  sampleRate: number;
  duration: number;
  peakAmplitude: number;
}

/**
 * Audio context manager singleton
 */
class ResonatorEngine {
  private module: ResonatorModule | null = null;
  private audioContext: AudioContext | null = null;
  private isInitialized = false;

  /**
   * Initialize the WASM module
   */
  async initialize(): Promise<void> {
    if (this.isInitialized) return;

    try {
      // Dynamic import of WASM module
      const wasmModule = await import('../wasm/resonator.js');
      this.module = await wasmModule.default() as ResonatorModule;
      
      // Initialize audio context
      this.audioContext = new (window.AudioContext || (window as unknown as { webkitAudioContext: typeof AudioContext }).webkitAudioContext)({
        sampleRate: 48000,
        latencyHint: 'interactive',
      });

      this.isInitialized = true;
      console.log('[ResonatorEngine] Initialized successfully');
    } catch (error) {
      console.error('[ResonatorEngine] Initialization failed:', error);
      throw new Error('Failed to initialize audio engine');
    }
  }

  /**
   * Convert preset to flat format for C++ engine
   */
  private presetToString(preset: Partial<Preset>): string {
    const params = [
      `duration_sec=${preset.durationSec ?? 480}`,
      `carrier_hz=${preset.carrierHz ?? 20.51}`,
      `gamma_hz=${preset.gammaHz ?? 90.12}`,
      `delta_hz=${preset.deltaHz ?? 1.25}`,
      `theta_hz=${preset.thetaHz ?? 5.08}`,
      `lower_hz=${preset.lowerHz ?? 10.55}`,
      `upper_hz=${preset.upperHz ?? 33.18}`,
      `infralow_hz=${preset.infralowHz ?? 0.125}`,
      `carrier_mix=${preset.carrierMix ?? 0.7}`,
      `harmonic_mix=${preset.harmonicMix ?? 0.3}`,
      `harmonic_decay=${preset.harmonicDecay ?? 1.6}`,
      `harmonics=${preset.harmonics ?? 4}`,
      `stereo_width=${preset.stereoWidth ?? 0.5}`,
      `stereo_expansion=${preset.stereoExpansion ?? 1.0}`,
      `headroom=${preset.headroom ?? 0.3}`,
      `mod_depth=${preset.modDepth ?? 0.25}`,
      `gamma_intensity=${preset.gammaIntensity ?? 0.6}`,
      `modulation_mult=${preset.modulationMult ?? 1.0}`,
      `exit_curve=${preset.exitCurve ?? 0.5}`,
      `attack_percent=${preset.attackPercent ?? 5}`,
      `release_percent=${preset.releasePercent ?? 5}`,
      `mode=${preset.mode ?? 'drift'}`,
    ];

    return params.join('\n');
  }

  /**
   * Render audio to buffers using WASM engine
   */
  async render(preset: Partial<Preset>): Promise<RenderResult> {
    if (!this.isInitialized || !this.module) {
      throw new Error('Engine not initialized');
    }

    const presetStr = this.presetToString(preset);
    
    try {
      // Allocate memory for preset string
      const strLen = this.module.lengthBytesUTF8(presetStr) + 1;
      const presetPtr = this.module._wasm_malloc(strLen);
      
      if (!presetPtr) {
        throw new Error('Failed to allocate memory for preset');
      }

      // Copy string to WASM memory
      this.module.stringToUTF8(presetStr, presetPtr, strLen);

      // Create engine instance
      const enginePtr = this.module._engine_create(presetPtr);
      this.module._wasm_free(presetPtr);

      if (!enginePtr) {
        throw new Error('Failed to create engine instance');
      }

      // Render audio
      const resultPtr = this.module._engine_render(enginePtr);
      this.module._engine_free(enginePtr);

      if (!resultPtr) {
        throw new Error('Render failed');
      }

      // Extract results
      const sampleRate = this.module._result_get_sample_rate(resultPtr);
      const numSamples = this.module._result_get_num_samples(resultPtr);
      const leftBufferPtr = this.module._result_get_left_buffer(resultPtr);
      const rightBufferPtr = this.module._result_get_right_buffer(resultPtr);
      const peak = this.module._result_get_peak(resultPtr);
      const duration = this.module._result_get_duration(resultPtr);

      // Copy buffers from WASM memory
      const leftChannel = new Float32Array(
        this.module.HEAPF32.buffer,
        leftBufferPtr,
        numSamples
      ).slice();

      const rightChannel = new Float32Array(
        this.module.HEAPF32.buffer,
        rightBufferPtr,
        numSamples
      ).slice();

      // Free result
      this.module._result_free(resultPtr);

      return {
        leftChannel,
        rightChannel,
        sampleRate,
        duration,
        peakAmplitude: peak,
      };
    } catch (error) {
      console.error('[ResonatorEngine] Render error:', error);
      throw error;
    }
  }

  /**
   * Play rendered audio through speakers
   */
  async play(renderResult: RenderResult): Promise<void> {
    if (!this.audioContext) {
      throw new Error('Audio context not initialized');
    }

    // Resume context if suspended
    if (this.audioContext.state === 'suspended') {
      await this.audioContext.resume();
    }

    const { leftChannel, rightChannel, sampleRate } = renderResult;
    
    // Create stereo buffer
    const buffer = this.audioContext.createBuffer(2, leftChannel.length, sampleRate);
    buffer.copyToChannel(leftChannel, 0);
    buffer.copyToChannel(rightChannel, 1);

    // Create source and gain nodes
    const source = this.audioContext.createBufferSource();
    source.buffer = buffer;

    const gainNode = this.audioContext.createGain();
    gainNode.gain.value = 0.5; // Master volume

    // Connect graph
    source.connect(gainNode);
    gainNode.connect(this.audioContext.destination);

    // Play
    source.start(0);

    return new Promise((resolve) => {
      source.onended = () => resolve();
    });
  }

  /**
   * Export rendered audio as WAV blob
   */
  exportWAV(renderResult: RenderResult): Blob {
    const { leftChannel, rightChannel, sampleRate } = renderResult;
    const numSamples = leftChannel.length;
    const numChannels = 2;
    const bytesPerSample = 2;
    const blockAlign = numChannels * bytesPerSample;
    const byteRate = sampleRate * blockAlign;
    const dataSize = numSamples * blockAlign;

    // WAV header (44 bytes)
    const header = new ArrayBuffer(44);
    const view = new DataView(header);

    // RIFF chunk
    this.writeString(view, 0, 'RIFF');
    view.setUint32(4, 36 + dataSize, true);
    this.writeString(view, 8, 'WAVE');

    // fmt chunk
    this.writeString(view, 12, 'fmt ');
    view.setUint32(16, 16, true); // Subchunk1Size
    view.setUint16(20, 1, true); // AudioFormat (PCM)
    view.setUint16(22, numChannels, true);
    view.setUint32(24, sampleRate, true);
    view.setUint32(28, byteRate, true);
    view.setUint16(32, blockAlign, true);
    view.setUint16(34, 16, true); // BitsPerSample

    // data chunk
    this.writeString(view, 36, 'data');
    view.setUint32(40, dataSize, true);

    // Convert float to 16-bit PCM
    const pcmData = new Int16Array(numSamples * numChannels);
    for (let i = 0; i < numSamples; i++) {
      // Left channel
      let sample = leftChannel[i] * 32767;
      sample = Math.max(-32768, Math.min(32767, sample));
      pcmData[i * 2] = sample;

      // Right channel
      sample = rightChannel[i] * 32767;
      sample = Math.max(-32768, Math.min(32767, sample));
      pcmData[i * 2 + 1] = sample;
    }

    // Combine header and data
    const wavBuffer = new Uint8Array(header.byteLength + pcmData.byteLength);
    wavBuffer.set(new Uint8Array(header), 0);
    wavBuffer.set(new Uint8Array(pcmData.buffer), header.byteLength);

    return new Blob([wavBuffer], { type: 'audio/wav' });
  }

  /**
   * Write string to DataView
   */
  private writeString(view: DataView, offset: number, string: string): void {
    for (let i = 0; i < string.length; i++) {
      view.setUint8(offset + i, string.charCodeAt(i));
    }
  }

  /**
   * Get current audio context time
   */
  getCurrentTime(): number {
    return this.audioContext?.currentTime ?? 0;
  }

  /**
   * Check if audio context is running
   */
  isRunning(): boolean {
    return this.audioContext?.state === 'running';
  }

  /**
   * Suspend audio context
   */
  async suspend(): Promise<void> {
    await this.audioContext?.suspend();
  }

  /**
   * Resume audio context
   */
  async resume(): Promise<void> {
    await this.audioContext?.resume();
  }

  /**
   * Clean up resources
   */
  dispose(): void {
    this.audioContext?.close();
    this.audioContext = null;
    this.module = null;
    this.isInitialized = false;
  }
}

 * Preset engine for managing frequency configurations
 */
export class PresetEngine {
  /**
   * Calculate derived frequencies based on base frequency
   */
  static calculateHarmonics(baseFreq: number, count: number, decay: number): number[] {
    const PHI = 1.618033988749895;
    const harmonics: number[] = [];
    
    for (let i = 1; i <= count; i++) {
      const freq = baseFreq * Math.pow(PHI, i);
      const amplitude = Math.pow(1 / decay, i);
      harmonics.push({ freq, amplitude } as unknown as number);
    }
    
    return harmonics;
  }

  /**
   * Validate preset parameters
   */
  static validatePreset(preset: Partial<Preset>): { valid: boolean; errors: string[] } {
    const errors: string[] = [];

    if (preset.carrierHz && (preset.carrierHz < 0.1 || preset.carrierHz > 20000)) {
      errors.push('Carrier frequency must be between 0.1 Hz and 20 kHz');
    }

    if (preset.durationSec && (preset.durationSec < 1 || preset.durationSec > 3600)) {
      errors.push('Duration must be between 1 and 3600 seconds');
    }

    if (preset.harmonics && (preset.harmonics < 1 || preset.harmonics > 16)) {
      errors.push('Harmonics must be between 1 and 16');
    }

    if (preset.stereoWidth && (preset.stereoWidth < 0 || preset.stereoWidth > 1)) {
      errors.push('Stereo width must be between 0 and 1');
    }

    return { valid: errors.length === 0, errors };
  }

  /**
   * Generate preset description based on parameters
   */
  static generateDescription(preset: Partial<Preset>): string {
    const parts: string[] = [];
    
    if (preset.mode) {
      parts.push(`${preset.mode.charAt(0).toUpperCase() + preset.mode.slice(1)} mode`);
    }
    
    if (preset.carrierHz) {
      parts.push(`carrier ${preset.carrierHz.toFixed(2)} Hz`);
    }
    
    if (preset.durationSec) {
      const mins = Math.floor(preset.durationSec / 60);
      parts.push(`${mins} min duration`);
    }
    
    return parts.join(', ');
  }
}

/**

export async function POST(request: NextRequest): Promise<NextResponse> {
  try {

    if (!session?.user?.id) {
      return NextResponse.json(
        { error: 'Authentication required' },
        { status: 401 }
      );
    }

    const body = await request.json();
    const { presetId } = body;

    if (!presetId) {
      return NextResponse.json(
        { error: 'Preset ID required' },
        { status: 400 }
      );
    }

    // Verify preset exists and user has access
    const preset = await prisma.preset.findUnique({
      where: { id: presetId },
    });

    if (!preset) {
      return NextResponse.json(
        { error: 'Preset not found' },
        { status: 404 }
      );
    }

    if (!preset.isPublic && preset.userId !== session.user.id) {
      return NextResponse.json(
        { error: 'Access denied' },
        { status: 403 }
      );

      data: {
        userId: session.user.id,
        presetId: preset.id,
        status: 'pending',

'use client';

import { useState } from 'react';
import { signIn } from 'next-auth/react';
import { useRouter, useSearchParams } from 'next/navigation';
import { motion } from 'framer-motion';
import { Button } from '@/components/ui/Button';
import { Activity, Github } from 'lucide-react';

export default function LoginPage() {
  const router = useRouter();
  const searchParams = useSearchParams();
  const callbackUrl = searchParams.get('callbackUrl') || '/dashboard';
  
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setIsLoading(true);
    setError(null);

    try {
      const result = await signIn('credentials', {
        email,
        password,
        redirect: false,
        callbackUrl,
      });

      if (result?.error) {
        setError('Invalid email or password');
      } else if (result?.ok) {
        router.push(callbackUrl);
      }
    } catch {
      setError('An error occurred. Please try again.');
    } finally {
      setIsLoading(false);
    }
  };

  const handleGoogleSignIn = () => {
    signIn('google', { callbackUrl });
  };

  return (
    <div className="min-h-screen bg-quantum-void flex items-center justify-center p-4">
      <motion.div
        initial={{ opacity: 0, y: 20 }}
        animate={{ opacity: 1, y: 0 }}
        className="w-full max-w-md"
      >
        <div className="glass-panel p-8">
          <div className="flex justify-center mb-8">
            <div className="w-16 h-16 rounded-full border-2 border-quantum-cyan flex items-center justify-center">
              <Activity className="w-8 h-8 text-quantum-cyan" />
            </div>
          </div>

          <h1 className="text-2xl font-bold text-center mb-2 text-gradient">
            ∆Ω-RESONATOR

            Sign in to access quantum audio synthesis

            <div className="mb-6 p-3 bg-red-500/10 border border-red-500/30 rounded text-red-400 text-sm">
              {error}
            </div>
          )}

          <form onSubmit={handleSubmit} className="space-y-4">
            <div>
              <label className="block text-sm font-medium mb-2">Email</label>
              <input
                type="email"
                value={email}
                onChange={(e) => setEmail(e.target.value)}
                className="w-full bg-background border border-input rounded-md px-3 py-2 text-white focus:outline-none focus:ring-2 focus:ring-quantum-cyan"
                required
              />
            </div>

            <div>
              <label className="block text-sm font-medium mb-2">Password</label>
              <input
                type="password"
                value={password}
                onChange={(e) => setPassword(e.target.value)}
                className="w-full bg-background border border-input rounded-md px-3 py-2 text-white focus:outline-none focus:ring-2 focus:ring-quantum-cyan"
                required
              />

              type="submit"
              className="w-full h-11"

            >
              Sign In
            </Button>

#!/bin/bash
# @file scripts/build-wasm.sh
# @description Build WebAssembly module from C++ source

set -e

echo "Building ∆Ω-RESONATOR WASM module..."

# Check for Emscripten
if ! command -v emcc &> /dev/null; then
    echo "Error: Emscripten not found. Install from https://emscripten.org/"
    exit 1
fi

# Create output directory
mkdir -p public/wasm

# Build WASM with optimizations
emcc lib/wasm/resonator.cpp \
    -o public/wasm/resonator.js \
    -s WASM=1 \
    -s EXPORTED_FUNCTIONS='["_malloc", "_free", "_engine_create", "_engine_render", "_result_get_sample_rate", "_result_get_num_samples", "_result_get_left_buffer", "_result_get_right_buffer", "_result_get_peak", "_result_get_duration", "_result_free", "_engine_free"]' \

    -s MAXIMUM_MEMORY=256MB \
